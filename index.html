☁️ Supabase Cloud-Setup

Wenn du die App mit Supabase verbinden möchtest, kopiere das SQL unten und füge es im SQL Editor deines Supabase-Projekts ein. Danach trage Project-URL und Anon-Key hier ein.

-- ═══════════════════════════════════════════════════
-- HIGHDENT – Supabase Cloud-Setup
-- Dieses SQL im Supabase SQL Editor ausführen
-- ═══════════════════════════════════════════════════

-- 1. Tabelle für App-Daten (ein JSON-Dokument pro Gerät/User)
CREATE TABLE IF NOT EXISTS app_data (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  device_name TEXT DEFAULT 'Unbenannt',
  pin_hash TEXT,
  data JSONB NOT NULL DEFAULT '{}'::jsonb,
  version INTEGER DEFAULT 1,
  created_at TIMESTAMPTZ DEFAULT now(),
  updated_at TIMESTAMPTZ DEFAULT now()
);

-- 2. Tabelle für Bestandsaufnahme-Fotos (separat wegen Größe)
CREATE TABLE IF NOT EXISTS ba_photos (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  app_data_id UUID REFERENCES app_data(id) ON DELETE CASCADE,
  assessment_id BIGINT NOT NULL,
  section_id TEXT NOT NULL,
  item_key TEXT NOT NULL,
  photo_index INTEGER DEFAULT 0,
  photo_data TEXT NOT NULL,
  created_at TIMESTAMPTZ DEFAULT now()
);

-- 3. Sync-Log für Konfliktauflösung
CREATE TABLE IF NOT EXISTS sync_log (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  app_data_id UUID REFERENCES app_data(id) ON DELETE CASCADE,
  action TEXT NOT NULL,
  timestamp TIMESTAMPTZ DEFAULT now(),
  device_name TEXT,
  data_hash TEXT
);

-- 4. Auto-Update des updated_at Timestamps
CREATE OR REPLACE FUNCTION update_modified_column()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = now();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS update_app_data_modtime ON app_data;
CREATE TRIGGER update_app_data_modtime
  BEFORE UPDATE ON app_data
  FOR EACH ROW EXECUTE FUNCTION update_modified_column();

-- 5. Row Level Security aktivieren
ALTER TABLE app_data ENABLE ROW LEVEL SECURITY;
ALTER TABLE ba_photos ENABLE ROW LEVEL SECURITY;
ALTER TABLE sync_log ENABLE ROW LEVEL SECURITY;

-- 6. Alte Policies entfernen (falls vorhanden)
DROP POLICY IF EXISTS "Allow all access" ON app_data;
DROP POLICY IF EXISTS "Allow all access" ON ba_photos;
DROP POLICY IF EXISTS "Allow all access" ON sync_log;
DROP POLICY IF EXISTS "anon_full_access" ON app_data;
DROP POLICY IF EXISTS "anon_full_access" ON ba_photos;
DROP POLICY IF EXISTS "anon_full_access" ON sync_log;

-- 7. Neue Policies mit explizitem anon-Zugriff
CREATE POLICY "anon_full_access" ON app_data
  FOR ALL TO anon, authenticated USING (true) WITH CHECK (true);

CREATE POLICY "anon_full_access" ON ba_photos
  FOR ALL TO anon, authenticated USING (true) WITH CHECK (true);

CREATE POLICY "anon_full_access" ON sync_log
  FOR ALL TO anon, authenticated USING (true) WITH CHECK (true);

-- 7. Indizes für Performance
CREATE INDEX IF NOT EXISTS idx_app_data_updated
  ON app_data (updated_at DESC);

CREATE INDEX IF NOT EXISTS idx_ba_photos_assessment
  ON ba_photos (app_data_id, assessment_id);

CREATE INDEX IF NOT EXISTS idx_sync_log_app
  ON sync_log (app_data_id, timestamp DESC);

-- 9. Ersten Datensatz anlegen (nur wenn leer)
INSERT INTO app_data (device_name, data)
SELECT 'Highdent Hauptgerät', '{}'::jsonb
WHERE NOT EXISTS (SELECT 1 FROM app_data LIMIT 1);

-- ✅ Setup abgeschlossen!
-- Trage jetzt die Project-URL und den Anon-Key
-- in den Highdent-Einstellungen ein.
